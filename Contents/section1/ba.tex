\begin{itemize}
    \item \textbf{int}:
    \begin{itemize}
        \item $-2,147,483,648$ to $2,147,483,647$ (10 digits)
        \item $-2^{31}$ to $2^{31} - 1$
        \item $-10^9$ to $10^9$
    \end{itemize}

    \item \textbf{unsigned long long int}:
    \begin{itemize}
        \item Begins with 9, and has a total of 19 digits
        \item $2^{63} -1$
        \item $10^{18}$
    \end{itemize}

    \item \textbf{array}:
    \begin{itemize}
        \item Do not declare with a size larger than 30,005.
    \end{itemize}
\end{itemize}
